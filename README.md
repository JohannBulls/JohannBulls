3. BinarySearchServiceImpl.java

Implementaci√≥n de la b√∫squeda binaria de forma recursiva.

package com.proyecto.service;

import org.springframework.stereotype.Service;
import java.util.List;

@Service
public class BinarySearchServiceImpl implements SearchService {

    @Override
    public int linearSearch(List<String> list, String value) {
        throw new UnsupportedOperationException("Not implemented in BinarySearchServiceImpl");
    }

    @Override
    public int binarySearch(List<String> sortedList, String value) {
        return binarySearchHelper(sortedList, value, 0, sortedList.size() - 1);
    }

    private int binarySearchHelper(List<String> list, String value, int left, int right) {
        if (left > right) {
            return -1;
        }
        int mid = left + (right - left) / 2;
        int compareResult = list.get(mid).compareTo(value);

        if (compareResult == 0) {
            return mid;
        } else if (compareResult > 0) {
            return binarySearchHelper(list, value, left, mid - 1);
        } else {
            return binarySearchHelper(list, value, mid + 1, right);
        }
    }
}

6. RoundRobinProxy.java

Clase de proxy que distribuye las solicitudes de forma round-robin entre dos instancias de servicios.

package com.proyecto.proxy;

import org.springframework.beans.factory.annotation.Value;
import org.springframework.stereotype.Service;
import java.net.HttpURLConnection;
import java.net.URL;
import java.util.List;
import java.io.BufferedReader;
import java.io.InputStreamReader;

@Service
public class RoundRobinProxy {

    @Value("${service.instances}")
    private List<String> instances;

    private int currentIndex = 0;

    public String forwardRequest(String endpoint) throws Exception {
        String urlStr = instances.get(currentIndex) + endpoint;
        currentIndex = (currentIndex + 1) % instances.size();

        URL url = new URL(urlStr);
        HttpURLConnection conn = (HttpURLConnection) url.openConnection();
        conn.setRequestMethod("GET");

        BufferedReader in = new BufferedReader(new InputStreamReader(conn.getInputStream()));
        StringBuilder content = new StringBuilder();
        String line;
        while ((line = in.readLine()) != null) {
            content.append(line);
        }
        in.close();
        return content.toString();
    }
}

7. ProxyController.java

Controlador para recibir solicitudes en el proxy y delegarlas a los servicios backend.

package com.proyecto.controller;

import com.proyecto.proxy.RoundRobinProxy;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.web.bind.annotation.*;

@RestController
@RequestMapping("/proxy")
public class ProxyController {

    @Autowired
    private RoundRobinProxy proxy;

    @GetMapping("/linearsearch")
    public String linearSearch(@RequestParam String list, @RequestParam String value) {
        try {
            return proxy.forwardRequest("/linearsearch?list=" + list + "&value=" + value);
        } catch (Exception e) {
            return "Error forwarding request: " + e.getMessage();
        }
    }

    @GetMapping("/binarysearch")
    public String binarySearch(@RequestParam String list, @RequestParam String value) {
        try {
            return proxy.forwardRequest("/binarysearch?list=" + list + "&value=" + value);
        } catch (Exception e) {
            return "Error forwarding request: " + e.getMessage();
        }
    }
}
